import Events from '../../static/file/events.json';

:::info
Ceci est une liste non exhaustive des événements disponibles avec [Minecraft Forge](http://minecraftforge.net/). Il vous est cependant possible de contribuer à cette liste. Pour cela vous pouvez proposer une modification de ce [fichier](https://github.com/Les-Moddeurs-Francais/Documentation-Forge/blob/master/static/file/events.json).
:::

export function EventList() {
    return (
        <div>
            {Events.category.map((category, index) => {
                return (
                    <div>
                        <h2 id={category.title.toLowerCase().replaceAll(" ", "-")}>{category.title}</h2>
                        {category.events.map((data, index) => {
                            return (
                                <details>
                                    <summary>
                                        <strong>{data.name}</strong>
                                    </summary>
                                    <div>
                                        <p>{data.description}</p>
                                        <strong>
                                            Annulable : {data.cancelable === true ? '✔' : '❌'}
                                        </strong>
                                    </div>
                                </details>
                            )
                        })}
                    </div>
                )
            })}
        </div>
    )
}
